༄༅། །རྒྱུད་སྡེ་ཁག་གཅིག་ཕྱོགས་བསྒྲིགས་ཀྱི་དཀར་ཆག། ༠༠༡ ཚུལ་དང་པོ་གསུམ་གྱི་དོན་ཉུང་ངུའི་ངག་གིས་སྟོན་པར་བྱེད་པའི་ལེགས་པར་བཤད་པ་ཚུལ་གསུམ་གསལ་བར་བྱེད་པའི་རིན་པོ་ཆེའི་རྒྱན། ༠༠༢ རྒྱུད་ཀྱི་མངོན་པར་རྟོགས་པའི་དོན་ཉུང་ངུའི་ངག་གིས་གཏན་ལ་འབེབས་པའི་བསྟན་བཅོས་ལྗོན་ཤིང་མཛེས་པར་བྱེད་པའི་རྒྱན། ༠༠༣ འཇམ་དཔལ་ཡེ་ཤེས་སེམས་དཔའི་དོན་དམ་པའི་མཚན་ཡང་དག་པར་བརྗོད་པ། ༠༠༤ དབང་མདོར་བསྟན་པ་བཞུགས་སོ། ༠༠༥ ཀྱེའི་རྡོ་རྗེ་ཞེས་བྱ་བ་རྒྱུད་ཀྱི་རྒྱལ་པོ་བཞུགས་སོ། ༠༠༦ འཕགས་པ་མཁའ་འགྲོ་མ་རྡོ་རྗེ་གུར་ཞེས་བྱ་བ་བརྟག་པ་གཅིག་པ་ལེའུ་བཅོ་ལྔ་པ་ཤླཽ་ཀ་སྟོང་དང་ཉི་ཤུ་ལྔ་པ། ༠༠༧ དཔལ་རྡོ་རྗེ་འཇིགས་བྱེད་ཀྱི་རྟོག་པའི་རྒྱུད་ཀྱི་རྒྱལ་པོ། ༠༠༨ ཡང་དག་པར་སྦྱོར་བ་ཞེས་བྱ་བའི་རྒྱུད་ཆེན་པོ། ༠༠༩ མའི་མ་བླ་ན་མེད་པ་རྣལ་འབྱོར་གསང་མཐའི་རྒྱུད་ཀྱི་རྒྱལ་པོ་བདེ་མཆོག་ཉུང་ངུ་། ༠༡༠ དཔལ་སངས་རྒྱས་ཐམས་ཅད་དང་མཉམ་པར་སྦྱོར་བ་མཁའ་འགྲོ་མ་སྒྱུ་མ་བདེ་བའི་མཆོག་ཅེས་བྱ་བའི་རྒྱུད་ཕྱི་མ། ༠༡༡ དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་སྐུ་གསུང་ཐུགས་ཀྱི་གསང་ཆེན་གསང་བ་འདུས་པ་ཞེས་བྱ་བ་བརྟག་པའི་རྒྱལ་པོ་ཆེན་པོ། ༠༡༢ དཔལ་རྡོ་རྗེ་འཇིགས་བྱེད་ཆེན་པོའི་རྒྱུད་ཅེས་བྱ་བ། ༠༡༣ རྡོ་རྗེ་ཕུར་པ་རྩ་བའི་རྒྱུད་ཀྱི་དུམ་བུ། ༠༡༤ དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ངན་སོང་ཐམས་ཅད་ཡོངས་སུ་སྦྱོང་བ་གཟི་བརྗིད་ཀྱི་རྒྱལ་པོའི་བརྟག་པ་བཅས་བཞུགས་སོ།། PAGE \* MERGEFORMAT 2 དཀར་ཆག རྒྱུད་སྡེ་ཁག་གཅིག་ཕྱོགས་བསྒྲིགས་ཀྱི་དཀར་ཆག PAGE \* MERGEFORMAT 1
